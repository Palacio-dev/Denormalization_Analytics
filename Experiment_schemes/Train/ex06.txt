<NORMALIZED MODEL>
    DROP TABLE IF EXISTS UF CASCADE;

    CREATE TABLE UF(
        id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
        sigla_uf VARCHAR(2) UNIQUE NOT NULL,
        nome VARCHAR(100) NOT NULL,
        qtd_habitantes INT CHECK(qtd_habitantes >= 0)
    );

    DROP TABLE IF EXISTS MUNICIPIO CASCADE;

    CREATE TABLE MUNICIPIO(
        id_municipio INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
        id_uf INT NOT NULL REFERENCES UF (id) ON DELETE CASCADE,
        nome_municipio VARCHAR(100) NOT NULL,
        qtd_IES INT CHECK (qtd_IES >= 0),
        qtd_ESCOLAS INT (qtd_ESCOLAS >= 0),
        media_ideb REAL,
        media_saeb REAL,
        media_enade REAL,
        total_participantes INT CHECK (total_participantes >= 0)
    );

    DROP TABLE IF EXISTS IDEB CASCADE;

    CREATE TABLE IDEB(
        id_ideb INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
        rend_1 REAL,
        rend_2 REAL,
        rend_3 REAL,
        rend_4 REAL,
        ind_rend REAL,
        nota_ideb REAL
    );

    DROP TABLE IF EXISTS ENADE CASCADE;

    CREATE TABLE ENADE(
        id_enade INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
        total_inscritos INT CHECK (total_inscritos >= 0),
        total_concluintes INT CHECK (total_concluintes >= 0),
        nota_bruta_ce REAL,
        nota_padronizada_ce REAL,
        nota_bruta_fg REAL,
        nota_padronizada_fg REAL,
        nota_enade_continua REAL,
        nota_enade_faixa REAL
    );

    DROP TABLE IF EXISTS SAEB CASCADE;

    CREATE TABLE SAEB(
        id_saeb INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
        nota_mat REAL,
        nota_port REAL,
        nota_padrao REAL
    );

    DROP TABLE IF EXISTS IES CASCADE;

    CREATE TABLE IES(
        id_ies INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
        id_municipio INT NOT NULL REFERENCES MUNICIPIO(id_municipio) ON DELETE CASCADE,
        nome_ies VARCHAR(100) NOT NULL,
        sigla_ies VARCHAR(20),
        codigo_ies INT UNIQUE NOT NULL,
        rede VARCHAR(100) NOT NULL
    );

    DROP TABLE IF EXISTS ESCOLA CASCADE;

    CREATE TABLE ESCOLA(
        id_escola INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
        id_municipio INT NOT NULL REFERENCES MUNICIPIO(id_municipio) ON DELETE CASCADE,
        nome_escola VARCHAR(100) NOT NULL,
        cod_escola INTEGER UNIQUE NOT NULL,
        rede VARCHAR(100) NOT NULL
    );

    DROP TABLE IF EXISTS CURSO CASCADE;

    CREATE TABLE CURSO(
        id_curso INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
        id_IES INT NOT NULL REFERENCES IES(id_IES) ON DELETE CASCADE,
        nome_curso VARCHAR(100) NOT NULL
    );

    DROP TABLE IF EXISTS ANO CASCADE; 

    CREATE TABLE ANO(
        id_ano INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
        id_curso INT NOT NULL REFERENCES curso(id_curso) ON DELETE CASCADE,
        id_enade INT NOT NULL REFERENCES enade(id_enade) ON DELETE CASCADE,
        id_saeb INT NOT NULL REFERENCES saeb(id_saeb) ON DELETE CASCADE,
        id_ideb INT NOT NULL REFERENCES ideb(id_ideb) ON DELETE CASCADE,
        ano INT NOT NULL
    );
</NORMALIZED MODEL>


<DENORMALIZED MODEL>
    CREATE TABLE Municipio (
        SIGLA_UF VARCHAR(2),
        NOME_UF VARCHAR(50),
        NOME_MUNICIPIO VARCHAR(100),
        PRIMARY KEY (SIGLA_UF, NOME_MUNICIPIO)
    );

    CREATE TABLE Escola (
        NOME_ESCOLA VARCHAR(200),
        CODIGO_ESCOLA VARCHAR(20),
        REDE_ESCOLA VARCHAR(20),
        ANO_ESCOLA INTEGER,
        IDEB_REND1 DOUBLE,
        IDEB_REND2 DOUBLE,
        IDEB_REND3 DOUBLE,
        IDEB_REND4 DOUBLE,
        IDEB_NOTA DOUBLE,
        SAEB_NOTA_MAT DOUBLE,
        SAEB_NOTA_PORT DOUBLE,
        SAEB_NOTA_PADRAO DOUBLE,
        SIGLA_UF VARCHAR(2),
        NOME_MUNICIPIO VARCHAR(100),
        PRIMARY KEY (CODIGO_ESCOLA, ANO_ESCOLA),
        FOREIGN KEY (SIGLA_UF, NOME_MUNICIPIO) REFERENCES Municipio(SIGLA_UF, NOME_MUNICIPIO)
    );
            
    CREATE TABLE Curso (
        NOME_IES VARCHAR(200),
        SIGLA_IES VARCHAR(20),
        CODIGO_IES VARCHAR(20),
        NOME_CURSO VARCHAR(200),
        TOTAL_INSCRITOS INTEGER,
        TOTAL_CONCLUINTES INTEGER,
        NOTA_BRUTA_CE DOUBLE,
        NOTA_PADRONIZADA_CE DOUBLE,
        NOTA_BRUTA_FG DOUBLE,
        NOTA_PADRONIZADA_FG DOUBLE,
        NOTA_ENADE_CONTINUA DOUBLE,
        NOTA_ENADE_FAIXA INTEGER,
        ANO_ENADE INTEGER,
        SIGLA_UF VARCHAR(2),
        NOME_MUNICIPIO VARCHAR(100),
        PRIMARY KEY (CODIGO_IES, NOME_CURSO, ANO_ENADE),
        FOREIGN KEY (SIGLA_UF, NOME_MUNICIPIO) REFERENCES Municipio(SIGLA_UF, NOME_MUNICIPIO)
    );
</DENORMALIZED MODEL>


1. Creating a Municipio table

Techniques Used: 3 - Collapsing Relations

In this table, the one-to-many relationship between Municipio and UF is broken, this way, we have a big "Municipio" table
that brigs together the information about the UF and the Municipio.

2. Creating a Escola table

Techniques Used: 3 - Collapsing Relations and 1 - Adding Redundant Attributes.

The new Escola table brigs together the information previously had by the SAEB table and the IDEB table and ANO table, 
this way, the result of JOIN operations are now together in one big table, making redundant attibutes. Futhermore, we 
do not lose any information on the process of transformation, which is very important


3. Creating a Curso table


The new Curso table unites the information from the IES table and the ENADE table.


It is important to notice the relationships between the tables in the denormalized model. One "Escola" is linked to one 
"Municipio", it is a one-to-one mandatory relation. Other than that, one "Municipio" has many "Cursos", it is a one-to-many
relationship. This way, we do not lose any information on the denormalized model, we are just adding redudancy to gain 
speed in the process of gathering information.
