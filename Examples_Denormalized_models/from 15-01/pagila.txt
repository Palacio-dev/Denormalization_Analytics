BEGIN;

-- 1. VERTICAL PARTITIONING: Separating heavy text fields from the main film table to improve cache efficiency for analytical queries.
CREATE TABLE IF NOT EXISTS public.film_text
(
    film_id integer NOT NULL,
    description text COLLATE pg_catalog."default",
    special_features text[] COLLATE pg_catalog."default",
    fulltext tsvector NOT NULL,
    CONSTRAINT film_text_pkey PRIMARY KEY (film_id)
);

-- 2. COLLAPSING RELATIONS & DERIVED ATTRIBUTES: Denormalized film table merging language, category, and actor data, plus precomputed aggregates.
CREATE TABLE IF NOT EXISTS public.film
(
    film_id serial NOT NULL,
    title text COLLATE pg_catalog."default" NOT NULL,
    release_year year,
    language_name character(20) COLLATE pg_catalog."default" NOT NULL,
    original_language_name character(20) COLLATE pg_catalog."default",
    rental_duration smallint NOT NULL DEFAULT 3,
    rental_rate numeric(4, 2) NOT NULL DEFAULT 4.99,
    length smallint,
    replacement_cost numeric(5, 2) NOT NULL DEFAULT 19.99,
    rating mpaa_rating DEFAULT 'G'::mpaa_rating,
    categories text[] COLLATE pg_catalog."default",
    actors text[] COLLATE pg_catalog."default",
    total_rentals integer NOT NULL DEFAULT 0,
    total_revenue numeric(10, 2) NOT NULL DEFAULT 0.00,
    last_update timestamp with time zone NOT NULL DEFAULT now(),
    CONSTRAINT film_pkey PRIMARY KEY (film_id)
);

-- 3. COLLAPSING RELATIONS: Denormalized store table merging address, city, and country.
CREATE TABLE IF NOT EXISTS public.store
(
    store_id serial NOT NULL,
    manager_staff_id integer NOT NULL,
    address text COLLATE pg_catalog."default" NOT NULL,
    address2 text COLLATE pg_catalog."default",
    district text COLLATE pg_catalog."default" NOT NULL,
    city text COLLATE pg_catalog."default" NOT NULL,
    country text COLLATE pg_catalog."default" NOT NULL,
    postal_code text COLLATE pg_catalog."default",
    phone text COLLATE pg_catalog."default" NOT NULL,
    last_update timestamp with time zone NOT NULL DEFAULT now(),
    CONSTRAINT store_pkey PRIMARY KEY (store_id)
);

-- 4. COLLAPSING RELATIONS & DERIVED ATTRIBUTES: Denormalized customer table merging address, city, and country, plus precomputed aggregates.
CREATE TABLE IF NOT EXISTS public.customer
(
    customer_id serial NOT NULL,
    store_id integer NOT NULL,
    first_name text COLLATE pg_catalog."default" NOT NULL,
    last_name text COLLATE pg_catalog."default" NOT NULL,
    email text COLLATE pg_catalog."default",
    activebool boolean NOT NULL DEFAULT true,
    create_date date NOT NULL DEFAULT CURRENT_DATE,
    active integer,
    address text COLLATE pg_catalog."default" NOT NULL,
    address2 text COLLATE pg_catalog."default",
    district text COLLATE pg_catalog."default" NOT NULL,
    city text COLLATE pg_catalog."default" NOT NULL,
    country text COLLATE pg_catalog."default" NOT NULL,
    postal_code text COLLATE pg_catalog."default",
    phone text COLLATE pg_catalog."default" NOT NULL,
    total_rentals integer NOT NULL DEFAULT 0,
    total_spent numeric(10, 2) NOT NULL DEFAULT 0.00,
    last_update timestamp with time zone DEFAULT now(),
    CONSTRAINT customer_pkey PRIMARY KEY (customer_id)
);

-- 5. COLLAPSING RELATIONS: Denormalized staff table merging address, city, and country.
CREATE TABLE IF NOT EXISTS public.staff
(
    staff_id serial NOT NULL,
    first_name text COLLATE pg_catalog."default" NOT NULL,
    last_name text COLLATE pg_catalog."default" NOT NULL,
    email text COLLATE pg_catalog."default",
    store_id integer NOT NULL,
    active boolean NOT NULL DEFAULT true,
    username text COLLATE pg_catalog."default" NOT NULL,
    password text COLLATE pg_catalog."default",
    picture bytea,
    address text COLLATE pg_catalog."default" NOT NULL,
    address2 text COLLATE pg_catalog."default",
    district text COLLATE pg_catalog."default" NOT NULL,
    city text COLLATE pg_catalog."default" NOT NULL,
    country text COLLATE pg_catalog."default" NOT NULL,
    postal_code text COLLATE pg_catalog."default",
    phone text COLLATE pg_catalog."default" NOT NULL,
    last_update timestamp with time zone NOT NULL DEFAULT now(),
    CONSTRAINT staff_pkey PRIMARY KEY (staff_id)
);

-- 6. KEEPING NECESSARY TABLES: Inventory (Required for physical stock tracking per store).
CREATE TABLE IF NOT EXISTS public.inventory
(
    inventory_id serial NOT NULL,
    film_id integer NOT NULL,
    store_id integer NOT NULL,
    last_update timestamp with time zone NOT NULL DEFAULT now(),
    CONSTRAINT inventory_pkey PRIMARY KEY (inventory_id)
);

-- 7. COLLAPSING RELATIONS & REDUNDANT/DERIVED ATTRIBUTES: Fact table merging rentals and payments, pulling in film_id and store_id to avoid inventory joins.
CREATE TABLE IF NOT EXISTS public.rental_payment
(
    rental_payment_id serial NOT NULL,
    rental_id integer NOT NULL,
    payment_id integer,
    customer_id integer NOT NULL,
    staff_id integer NOT NULL,
    inventory_id integer NOT NULL,
    film_id integer NOT NULL,
    store_id integer NOT NULL,
    rental_date timestamp with time zone NOT NULL,
    return_date timestamp with time zone,
    payment_date timestamp with time zone,
    amount numeric(5, 2),
    rental_duration_hours numeric(10, 2),
    last_update timestamp with time zone NOT NULL DEFAULT now(),
    CONSTRAINT rental_payment_pkey PRIMARY KEY (rental_payment_id)
);

-- ADDING FOREIGN KEYS AND INDEXES FOR THE DENORMALIZED SCHEMA

ALTER TABLE IF EXISTS public.film_text
    ADD CONSTRAINT film_text_film_id_fkey FOREIGN KEY (film_id)
    REFERENCES public.film (film_id) MATCH SIMPLE
    ON UPDATE CASCADE
    ON DELETE CASCADE;

ALTER TABLE IF EXISTS public.store
    ADD CONSTRAINT store_manager_staff_id_fkey FOREIGN KEY (manager_staff_id)
    REFERENCES public.staff (staff_id) MATCH SIMPLE
    ON UPDATE CASCADE
    ON DELETE RESTRICT;

ALTER TABLE IF EXISTS public.customer
    ADD CONSTRAINT customer_store_id_fkey FOREIGN KEY (store_id)
    REFERENCES public.store (store_id) MATCH SIMPLE
    ON UPDATE CASCADE
    ON DELETE RESTRICT;
CREATE INDEX IF NOT EXISTS idx_fk_customer_store_id ON public.customer(store_id);

ALTER TABLE IF EXISTS public.staff
    ADD CONSTRAINT staff_store_id_fkey FOREIGN KEY (store_id)
    REFERENCES public.store (store_id) MATCH SIMPLE
    ON UPDATE NO ACTION
    ON DELETE NO ACTION;
CREATE INDEX IF NOT EXISTS idx_fk_staff_store_id ON public.staff(store_id);

ALTER TABLE IF EXISTS public.inventory
    ADD CONSTRAINT inventory_film_id_fkey FOREIGN KEY (film_id)
    REFERENCES public.film (film_id) MATCH SIMPLE
    ON UPDATE CASCADE
    ON DELETE RESTRICT;
CREATE INDEX IF NOT EXISTS idx_fk_inventory_film_id ON public.inventory(film_id);

ALTER TABLE IF EXISTS public.inventory
    ADD CONSTRAINT inventory_store_id_fkey FOREIGN KEY (store_id)
    REFERENCES public.store (store_id) MATCH SIMPLE
    ON UPDATE CASCADE
    ON DELETE RESTRICT;
CREATE INDEX IF NOT EXISTS idx_fk_inventory_store_id ON public.inventory(store_id);

ALTER TABLE IF EXISTS public.rental_payment
    ADD CONSTRAINT rental_payment_customer_id_fkey FOREIGN KEY (customer_id)
    REFERENCES public.customer (customer_id) MATCH SIMPLE
    ON UPDATE CASCADE
    ON DELETE RESTRICT;
CREATE INDEX IF NOT EXISTS idx_fk_rp_customer_id ON public.rental_payment(customer_id);

ALTER TABLE IF EXISTS public.rental_payment
    ADD CONSTRAINT rental_payment_staff_id_fkey FOREIGN KEY (staff_id)
    REFERENCES public.staff (staff_id) MATCH SIMPLE
    ON UPDATE CASCADE
    ON DELETE RESTRICT;
CREATE INDEX IF NOT EXISTS idx_fk_rp_staff_id ON public.rental_payment(staff_id);

ALTER TABLE IF EXISTS public.rental_payment
    ADD CONSTRAINT rental_payment_inventory_id_fkey FOREIGN KEY (inventory_id)
    REFERENCES public.inventory (inventory_id) MATCH SIMPLE
    ON UPDATE CASCADE
    ON DELETE RESTRICT;
CREATE INDEX IF NOT EXISTS idx_fk_rp_inventory_id ON public.rental_payment(inventory_id);

ALTER TABLE IF EXISTS public.rental_payment
    ADD CONSTRAINT rental_payment_film_id_fkey FOREIGN KEY (film_id)
    REFERENCES public.film (film_id) MATCH SIMPLE
    ON UPDATE CASCADE
    ON DELETE RESTRICT;
CREATE INDEX IF NOT EXISTS idx_fk_rp_film_id ON public.rental_payment(film_id);

ALTER TABLE IF EXISTS public.rental_payment
    ADD CONSTRAINT rental_payment_store_id_fkey FOREIGN KEY (store_id)
    REFERENCES public.store (store_id) MATCH SIMPLE
    ON UPDATE CASCADE
    ON DELETE RESTRICT;
CREATE INDEX IF NOT EXISTS idx_fk_rp_store_id ON public.rental_payment(store_id);

END;
