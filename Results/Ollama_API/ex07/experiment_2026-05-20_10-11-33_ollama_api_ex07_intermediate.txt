-- Updated Usina Table with Denormalized Fields
CREATE TABLE IF NOT EXISTS public."UsinaDenorm" (
    id SERIAL PRIMARY KEY,
    nome VARCHAR NOT NULL,
    potkw FLOAT,
    potfis FLOAT,
    iniciop DATE,
    atualizacao DATE,

    long DOUBLE PRECISION,
    estagio VARCHAR,

    proprietario VARCHAR NULL,       -- From Proprietarios

    id_rio INTEGER NOT NULL,
    rio VARCHAR NOT NULL,            -- From Rios

    id_localizacao INTEGER NOT NULL,
    municipio VARCHAR NOT NULL,      -- From Municipios
    uf CHAR(2) NOT NULL              -- From Estados
);

-- Copy original data into new table (you'd do this once after setup)
INSERT INTO public."UsinaDenorm"
SELECT 
    u.id, u.nome, u.potkw, u.potfis, u.iniciop, u.atualizacao,
    u.lat, u.long, u.estagio,
    u.id_proprietario, p.proprietario,
    u.id_rio, r.rio,
    u.id_localizacao, m.municipio, e.uf
FROM public."Usina" u
LEFT JOIN public."Proprietarios" p ON u.id_proprietario = p.id_proprietario
JOIN public."Rios" r ON u.id_rio = r.id_rio
JOIN public."Localizacoes" l ON u.id_localizacao = l.id_localização
JOIN public."Municipios" m ON l.id_municipio1 = m.id_municipio
JOIN public."Estados" e ON m.uf = e.uf;

-- Indexes Optimized for Read Queries
CREATE INDEX idx_usinadenorm_geo ON public."UsinaDenorm" (uf, municipio);
CREATE INDEX idx_usinadenorm_owner ON public."UsinaDenorm" (proprietario, nome);
CREATE INDEX idx_usinadenorm_capacity ON public."UsinaDenorm" (potkw DESC);
CREATE INDEX idx_usinadenorm_date ON public."UsinaDenorm" (atualizacao);

-- Materialized Summary Table for State-Level Reporting
CREATE MATERIALIZED VIEW public.usina_summary_by_state AS
SELECT
    uf,
    COUNT(*) AS total_usinas,
    SUM(potkw) AS total_potencia_kw,
    AVG(EXTRACT(YEAR FROM AGE(CURRENT_DATE, iniciop))) AS avg_age_years
FROM public."UsinaDenorm"
GROUP BY uf;

REFRESH MATERIALIZED VIEW CONCURRENTLY public.usina_summary_by_state;
