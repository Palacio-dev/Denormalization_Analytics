CREATE TABLE SALES_DATA_DENORMALIZED (
    -- Lineitem Columns
    L_ORDERKEY       INTEGER NOT NULL,
    L_LINENUMBER     INTEGER NOT NULL,
    L_QUANTITY       DECIMAL(15,2) NOT NULL,
    L_EXTENDEDPRICE  DECIMAL(15,2) NOT NULL,
    L_DISCOUNT       DECIMAL(15,2) NOT NULL,
    L_TAX            DECIMAL(15,2) NOT NULL,
    L_RETURNFLAG     CHAR(1) NOT NULL,
    L_LINESTATUS     CHAR(1) NOT NULL,
    L_SHIPDATE       DATE NOT NULL,
    L_COMMITDATE     DATE NOT NULL,
    L_RECEIPTDATE    DATE NOT NULL,
    L_SHIPINSTRUCT   CHAR(25) NOT NULL,
    L_SHIPMODE       CHAR(10) NOT NULL,

    -- Order Columns
    O_ORDERSTATUS    CHAR(1) NOT NULL,
    O_TOTALPRICE     DECIMAL(15,2) NOT NULL,
    O_ORDERDATE      DATE NOT NULL,
    O_ORDERPRIORITY  CHAR(15) NOT NULL,
    O_CLERK          CHAR(15) NOT NULL,
    O_SHIPPRIORITY   INTEGER NOT NULL,

    -- Part Columns
    P_PARTKEY        INTEGER NOT NULL,
    P_NAME           VARCHAR(55) NOT NULL,
    P_MFGR           CHAR(25) NOT NULL,
    P_BRAND          CHAR(10) NOT NULL,
    P_TYPE           VARCHAR(25) NOT NULL,
    P_SIZE           INTEGER NOT NULL,
    P_CONTAINER      CHAR(10) NOT NULL,
    P_RETAILPRICE    DECIMAL(15,2) NOT NULL,

    -- Supplier Columns
    S_SUPPKEY        INTEGER NOT NULL,
    S_NAME           CHAR(25) NOT NULL,
    S_ADDRESS        VARCHAR(40) NOT NULL,
    S_PHONE          CHAR(15) NOT NULL,
    S_ACCTBAL        DECIMAL(15,2) NOT NULL,

    -- PartSupp Columns
    PS_AVAILQTY      INTEGER NOT NULL,
    PS_SUPPLYCOST    DECIMAL(15,2) NOT NULL,

    -- Customer Columns
    C_CUSTKEY        INTEGER NOT NULL,
    C_NAME           VARCHAR(25) NOT NULL,
    C_ADDRESS        VARCHAR(40) NOT NULL,
    C_PHONE          CHAR(15) NOT NULL,
    C_ACCTBAL        DECIMAL(15,2) NOT NULL,
    C_MKTSEGMENT     CHAR(10) NOT NULL,

    -- Denormalized Customer Location Columns
    C_NATION_NAME    CHAR(25) NOT NULL,
    C_REGION_NAME    CHAR(25) NOT NULL,

    -- Denormalized Supplier Location Columns
    S_NATION_NAME    CHAR(25) NOT NULL,
    S_REGION_NAME    CHAR(25) NOT NULL
);
