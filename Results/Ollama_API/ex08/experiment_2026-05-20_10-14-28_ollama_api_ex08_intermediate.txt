### ➕ New Materialized View: `vw_matricula_analitica`

```sql
CREATE MATERIALIZED VIEW vw_matricula_analitica AS
SELECT 
    m.id_matricula,
    m.periodo,
    m.situacao_matricula,
    m.tipo_cota,
    m.ano,
    m.unidade_ensino,

    -- aluno info
    a.sexo,
    a.raca,
    a.portador_deficiencia,
    a.nacionalidade,

    -- curso info
    c.no_curso,
    c.tipo_curso,
    c.ds_tipo_nivel,
    c.ds_eixo_tecnologico,

    -- escola info
    e.tp_dependencia_adm_escola,
    e.inse,

    -- localizacao info
    l.uf,
    l.municipio
FROM matricula_ifb m
JOIN aluno_ifb a ON m.id_aluno = a.id_aluno
JOIN curso_ifb c ON m.co_curso = c.co_curso
JOIN escola e ON m.co_escola_educacenso = e.co_escola_educacenso
JOIN localizacao l ON e.id_loc = l.id_loc;
```

Indexes on materialized view:

```sql
CREATE INDEX idx_vw_matricula_uf ON vw_matricula_analitica(uf);
```

### ➕ Aggregate Table: `agg_escola_desempenho_anual`

```sql
CREATE TABLE agg_escola_desempenho_anual (
    nu_ano INTEGER NOT NULL,
    co_escola_educacenso VARCHAR(50),
    avg_nu_media_cn NUMERIC(7,2),
    avg_nu_media_ch NUMERIC(7,2),
    avg_nu_media_lp NUMERIC(7,2),
    avg_nu_media_mt NUMERIC(7,2),
    avg_nu_taxa_aprovacao NUMERIC(5,2),
    avg_nu_taxa_abandono NUMERIC(5,2),
    nu_taxa_participacao NUMERIC(5,2),
    count_alunos INTEGER,
    PRIMARY KEY(nu_ano, co_escola_educacenso)
);

-- Populate this table daily or hourly depending on freshness needs
INSERT INTO agg_escola_desempenho_anual
SELECT 
    de.nu_ano,
    de.co_escola_educacenso,
    AVG(de.nu_media_cn) AS avg_nu_media_cn,
    AVG(de.nu_media_ch) AS avg_nu_media_ch,
    AVG(de.nu_media_lp) AS avg_nu_media_lp,
    AVG(de.nu_media_mt) AS avg_nu_media_mt,
    AVG(ie.nu_taxa_aprovacao) AS avg_nu_taxa_aprovacao,
    AVG(ie.nu_taxa_abandono) AS avg_nu_taxa_abandono,
    MAX(pe.nu_taxa_participacao) AS nu_taxa_participacao,
    COUNT(DISTINCT m.id_aluno) AS count_alunos
FROM desempenho_enem de
JOIN indicadores_escolares ie ON de.co_escola_educacenso = ie.co_escola_educacenso AND de.nu_ano = ie.nu_ano
LEFT JOIN participacao_enem pe ON de.co_escola_educacenso = pe.co_escola_educacenso AND de.nu_ano = pe.nu_ano
JOIN matricula_ifb m ON de.co_escola_educacenso = m.co_escola_educacenso AND de.nu_ano::text = m.ano
GROUP BY de.nu_ano, de.co_escola_educacenso;
```

### ➕ Aggregate Table: `agg_aluno_empregabilidade`

```sql
CREATE TABLE agg_aluno_empregabilidade (
    sexo CHAR(1),
    raca VARCHAR(50),
    portador_deficiencia BOOLEAN,
    avg_remuneracao NUMERIC(10,2),
    avg_tempo_emprego INTEGER,
    total_vinculos_ativos INT,
    total_vinculos_total INT,
    PRIMARY KEY(sexo, raca, portador_deficiencia)
);

INSERT INTO agg_aluno_empregabilidade
SELECT 
    a.sexo,
    a.raca,
    a.portador_deficiencia,
    AVG(r.vl_remun_media_sm) AS avg_remuneracao,
    AVG(v.tempo_emprego) AS avg_tempo_emprego,
    SUM(CASE WHEN v.indicador_vinculo_ativo THEN 1 ELSE 0 END) AS total_vinculos_ativos,
    COUNT(*) AS total_vinculos_total
FROM vinculo_empregaticio v
JOIN aluno_ifb a ON v.id_aluno = a.id_aluno
JOIN remuneracao r ON v.id_remuneracao = r.id_remuneracao
GROUP BY a.sexo, a.raca, a.portador_deficiencia;
```
